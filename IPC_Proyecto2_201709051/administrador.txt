    private void loadTeams(BufferedReader br, String FileName) throws FileNotFoundException, IOException {
        reportNode reporteTemp = new reportNode(FileName);
        StringBuilder sb = new StringBuilder();
        String line = br.readLine();

        while (line != null) {
            sb.append(line);
            sb.append("\n");
            line = br.readLine();
        }
        line = sb.toString();
        String[] a = line.split("\n");
        for (String s : a) {
            String b[] = s.split("#");
            int team = 0, wins = 0;
            boolean error = false, duplicated = false;
            try {
                //No Blank Spaces
                for (String l : b) {
                    Scanner li = new Scanner(l);
                    String temp = li.next();
                }

                //Validate ID
                Scanner sc = new Scanner(b[0]);
                team = sc.nextInt();
                if (sc.hasNext()) {
                    throw new Exception();
                }
                try {
                    Equipo prueba = main.Teams.find("" + team);
                    if (prueba != null) {
                        reporteTemp.duplicated++;
                        duplicated = true;
                        throw new Exception();
                    }
                } catch (NullPointerException e) {

                }

                //Validate year
                Scanner scc = new Scanner(b[2]);
                String[] b2 = scc.next().split("/");
                Scanner sc2 = new Scanner(b2[0]);
                if (sc2.nextInt() > 31) {
                    throw new Exception("Error de dia");
                }
                Scanner sc3 = new Scanner(b2[1]);
                if (sc3.nextInt() > 12) {
                    throw new Exception("Error de mes");
                }
                Scanner sc4 = new Scanner(b2[2]);
                int year = sc4.nextInt();
                if (year > 2018 || year < 1000) {
                    throw new Exception("Error de año");
                }
                if (sc4.hasNext()) {
                    throw new Exception("Error de formato");
                }

                //Validate Name
                char q[] = b[1].toCharArray();
                for (char m : q) {
                    if (Character.isDigit(m) || Character.isLetter(m) || m == ' '); else {
                        throw new Exception();
                    }
                }

                //Validate Wins
                Scanner sc1 = new Scanner(b[3]);
                wins = sc1.nextInt();
                if (sc1.hasNext()) {
                    throw new Exception();
                }

            } catch (Exception e) {
                if (!duplicated) {
                    reporteTemp.invalid++;
                }
                error = true;
            }
            if (!error) {
                reporteTemp.succeded++;
                main.Teams.add(new Equipo("" + team, b[1], b[2], "" + wins));
                
            }
        }
        main.reports.add(reporteTemp);
    }

    private void loadPlayers(BufferedReader br, String FileName) throws FileNotFoundException, IOException {
        reportNode reporteTemp = new reportNode(FileName);
        StringBuilder sb = new StringBuilder();
        String line = br.readLine();

        while (line != null) {
            sb.append(line);
            sb.append("\n");
            line = br.readLine();
        }
        line = sb.toString();
        String[] a = line.split("\n");
        for (String s : a) {
            String b[] = s.split("#");
            boolean error = false, duplicated = false;
            int Team = 0, ID = 0, year = 0;
            String height = "", weight = "";
            try {
                //No Blank Spaces
                for (String l : b) {
                    Scanner li = new Scanner(l);
                    String temp = li.next();
                }

                //Validate team
                Scanner sc1 = new Scanner(b[6]);
                Team = sc1.nextInt();
                Equipo e = main.Teams.find("" + Team);

                //Validate ID
                Scanner sc = new Scanner(b[0]);
                ID = sc.nextInt();
                if (sc.hasNext()) {
                    throw new Exception();
                }
                try {
                    Jugador prueba = e.lista.find("" + ID);
                    if (prueba != null) {
                        duplicated = true;
                        reporteTemp.duplicated++;
                        throw new Exception();
                    }
                } catch (NullPointerException exe) {

                }

                //Validate Name
                char q[] = b[1].toCharArray();
                for (char m : q) {
                    if (Character.isLetter(m) || m == ' '); else {
                        throw new Exception();
                    }
                }

                //Validate Position
                Scanner po = new Scanner(b[2]);
                char[] v = po.next().toCharArray();
                if (!po.hasNext()) {
                    for (char m : v) {
                        if (!Character.isLetter(m)) {
                            throw new Exception();
                        }
                    }
                } else {
                    throw new Exception();
                }

                //Validate Birth Year
                Scanner sc3 = new Scanner(b[3]);
                year = sc3.nextInt();
                if (sc3.hasNext()) {
                    throw new Exception();
                }

                //Validate height
                DecimalFormat dF = new DecimalFormat();
                Scanner t = new Scanner(b[4]);
                double my = Double.valueOf(t.next());
                dF.setMaximumFractionDigits(2);
                height = dF.format(my);
                if (t.hasNext()) {
                    throw new Exception();
                }

                //Validate weight
                Scanner t2 = new Scanner(b[5]);
                double my2 = Double.valueOf(t2.next());
                weight = dF.format(my2);
                if (t2.hasNext()) {
                    throw new Exception();
                }
            } catch (Exception e) {
                if (!duplicated) {
                    reporteTemp.invalid++;
                }
                error = true;
            }
            if (!error) {
                reporteTemp.succeded++;
                Jugador jugador = new Jugador("" + year, b[1], b[2], "" + height, "" + weight, "" + Team, b[6], b[7]);
            }
        }
        main.reports.add(reporteTemp);
    }

    private void loadStickers(BufferedReader br, String FileName) throws FileNotFoundException, IOException {
        reportNode reporteTemp = new reportNode(FileName);
        StringBuilder sb = new StringBuilder();
        String line = br.readLine();

        while (line != null) {
            sb.append(line);
            sb.append("\n");
            line = br.readLine();
        }
        line = sb.toString();
        String[] a = line.split("\n");
        for (String s : a) {
            String b[] = s.split("#");
            boolean error = false, duplicated = false;
            int Team = 0, IDE = 0, wei = 0;
            int ID = 0;
            String ruta = null;
            try {

                //No Blank Spaces
                for (String l : b) {
                    Scanner li = new Scanner(l);
                    String temp = li.next();
                }

                //Validate ID
                Scanner sc4 = new Scanner(b[0]);
                ID = sc4.nextInt();
                if (sc4.hasNext()) {
                    throw new Exception();
                }
                if (ID < 0 || ID >= 1000) {
                    throw new Exception("ID Invalida");
                }
                try {
                    Estampa prueba = main.est.find("" + ID);
                    if (prueba != null) {
                        duplicated = true;
                        reporteTemp.duplicated++;
                        throw new Exception();
                    }
                } catch (NullPointerException e) {

                }

                //Validate team
                Scanner sc1 = new Scanner(b[2]);
                Team = sc1.nextInt();
                Equipo prueba = main.Teams.find("" + Team);

                //Validate IDJugador
                Scanner sc = new Scanner(b[1]);
                IDE = sc.nextInt();
                Jugador prueba2 = main.Teams.find("" + Team).lista.find("" + IDE);

                //Validate Rareza
                Scanner sc3 = new Scanner(b[3]);
                wei = sc3.nextInt();
                if (wei == 0 || wei == 1) {
                } else {
                    throw new Exception("Rareza invalida");
                }

                //Validate Ruta
                Scanner asdf = new Scanner(b[4]);
                ruta = asdf.next();
                File imagen = new File("src/" + ruta);
                if (!imagen.isFile() || !imagen.exists()) {
                    throw new Exception("imagen inexistente");
                }

            } catch (Exception e) {
                if (!duplicated) {
                    reporteTemp.invalid++;
                }
                error = true;
            }
            if (!error) {
                reporteTemp.succeded++;
                main.est.add(new Estampa("" + IDE, "" + Team, "" + wei, "/" + ruta));
            }
        }
        main.reports.add(reporteTemp);
    }
